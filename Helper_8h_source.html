180
181 const auto sumOfWeights =
182 std::accumulate(data.begin(), data.end(), 0u,
183 [](unsigned sum, const WeightedElement<T>& element) { return sum + element.weight; });
184
185 if (sumOfWeights == 0u)

192 size_t currentIdx = 0;
193
194 while (currentIdx < data.size())

196 currentSum += data[currentIdx].weight;
197 if (currentSum >= targetWeightValue)
198 break;
199 currentIdx++;
200 }
201
